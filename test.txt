it('GenerateTableRows(): should handle error by displaying service errors, stop loading, and dispatch LOADINGACTION(false)', () => {
  // Arrange
  const errorResponse = { error: { statusMessage: 'Server error occurred' } };

  const getReqSpy = spyOn(component.feeApiservice, 'getRequest')
    .and.returnValue(throwError(() => errorResponse));

  const displaySpy = spyOn(component as any, 'displayServiceErrors');
  const dispatchSpy = spyOn(component['store'], 'dispatch');

  component.loading = true;

  // Act
  component.GenerateTableRows();

  // Assert
  expect(getReqSpy).toHaveBeenCalledWith(component.appConfigService.getAllAltDivCostCenterMappingUrl);
  expect(displaySpy).toHaveBeenCalledWith('Server error occurred');
  expect(component.loading).toBeFalse();
  expect(dispatchSpy).toHaveBeenCalledWith(jasmine.any(LOADINGACTION));
});
