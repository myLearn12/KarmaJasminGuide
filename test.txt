it('InitialDataLoad() error case without statusMessage → still disables save, stops loading, dispatches action', () => {
  const utilsStub = {
    CreateRequiredFormControl: () => ({}),
    CreateDateFormControl: () => ({}),
    LoadComboBoxData: () => [],
  };
  const acctServiceStub = {
    LoadOfficerData: () => [],
    loadStatusData: () => [],
  };
  const storeMock = { dispatch: jasmine.createSpy('dispatch') };

  const component = new EditaccountComponent(
    {} as any,
    {} as any,
    storeMock as any,
    utilsStub as any,
    acctServiceStub as any
  );

  // feeApi throws without error.error.statusMessage
  component['feeApiservice'] = {
    getRequest: () => throwError(() => new Error('boom')),
  } as any;

  component['displayServiceErrors'] = jasmine.createSpy('displayServiceErrors');

  component['InitialDataLoad']();

  // even if no statusMessage, UI should flip to error state
  expect(component.disableSavebtn).toBeTrue();
  expect(component.loading).toBeFalse();
  expect(storeMock.dispatch).toHaveBeenCalled();
  // optional: ensure display called with something (don’t enforce exact text)
  expect(component['displayServiceErrors']).toHaveBeenCalled();
});
