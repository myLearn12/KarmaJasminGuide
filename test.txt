it('onSearchButtonClick() - other value & invalid string should call GenerateSearchAcctNameTableRows(url, "1")', () => {
  spyOn(component, 'GenerateSearchAcctNameTableRows').and.callFake(() => {});
  spyOn(store, 'dispatch').and.callFake(() => {});
  component.ValidateFormControls = jasmine.createSpy().and.returnValue(false);

  component._searchByRadioButton = { value: '3' };
  component.selectedSearchText = 'XYZ';
  component.utils = { isStringValid: () => false };
  component.appConfigService = { getAllAccountExportUrl: '/api/accounts/' };

  component.onSearchButtonClick();

  expect(component.loading).toBeTrue();
  expect(store.dispatch).toHaveBeenCalledWith(new LOADINGACTION(true));
  expect(component.GenerateSearchAcctNameTableRows)
    .toHaveBeenCalledWith('/api/accounts/', '1');
});
