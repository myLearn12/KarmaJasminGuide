it('revShareDivMappingMessage: sets mapping messages for division and alt division', () => {
  // Arrange
  (component as any).cbDivision = { value: '20' };
  (component as any).cbAltDivision = { value: '30' };

  component.divisions = [
    { divisionId: 20, divisionCd: 'D20' },
    { divisionId: 30, divisionCd: 'D30' }
  ] as any;

  component.revShareDivMapping = [
    { divisionId: 20, revShareDesc: 'RevShare-20' },
    { divisionId: 30, revShareDesc: 'RevShare-30' }
  ] as any;

  spyOn(component.utils, 'isStringValid').and.callFake((s: any) => !!s && s.trim() !== '');

  // Act
  component.revShareDivMappingMessage();

  // Assert
  expect(component.divCdMapMessage).toContain('Division Cd D20 is mapped to RevShare-20 revenue share.');
  expect(component.altDivCdMapMessage).toContain('Alt Division Cd D30 is mapped to RevShare-30 revenue share.');
});
