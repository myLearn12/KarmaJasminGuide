it('onUpdateModalClick(): else branch -> sets selectedunassignedaccountmap, calls setInitialValues, getRegionComboBoxValue and shows action modal', () => {
  // arrange: ensure component is in the path that runs the else branch
  component.isRowSelected = true;

  // give an example unassignedaccountmap object; method will JSON.stringify/parse it
  component.unassignedaccountmap = { regId: 10, accountNbr: 'A-123' };

  // spy the two internal helpers so we can assert they were called
  const setInitialSpy = spyOn(component as any, 'setIntialValues').and.callThrough();
  const getRegionSpy = spyOn(component as any, 'getRegionComboBoxValue').and.callThrough();

  // provide an actionModal object with a show method and spy it
  (component as any)._actionModal = { show: () => {} };
  const showSpy = spyOn((component as any)._actionModal, 'show').and.stub();

  // act
  component.onUpdateModalClick();

  // assert
  expect((component as any).selectedunassignedaccountmap).toEqual(JSON.parse(JSON.stringify(component.unassignedaccountmap)));
  expect(setInitialSpy).toHaveBeenCalled();
  expect(getRegionSpy).toHaveBeenCalled();
  expect(showSpy).toHaveBeenCalled();
});
