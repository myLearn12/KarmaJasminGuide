it('processRecord(): complete branch -> sets success modal title/message and shows modal', fakeAsync(() => {

  // Create a fake modal object so spyOn() works
  (component as any).successModal = { show: jasmine.createSpy('show') };

  // Mock feeApiService.putRequest to emit next() and complete()
  spyOn((component as any).feeApiservice, 'putRequest').and.returnValue(
    new Observable(observer => {
      observer.next({});
      observer.complete();
    })
  );

  // Spy store.dispatch
  const storeSpy = spyOn((component as any).store, 'dispatch');

  // Set an account number so message builds correctly
  (component as any).account = { accountNbr: 'ACCT123' };

  // Initial loading flag
  (component as any).loading = true;

  // Act
  component.processRecord();
  tick();

  // Assert
  expect((component as any).loading).toBeFalse();
  expect(storeSpy).toHaveBeenCalled();
  expect((component as any).successModalTitle).toBe('Reopen Account');
  expect((component as any).successModalMessage).toContain('ACCT123');
  expect((component as any).successModal.show).toHaveBeenCalled();
}));
