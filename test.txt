it('should hit error branch in GenerateSearchAcctNumTableRows() and set flags + call displayServiceErrors + dispatch LOADINGACTION(false)', () => {

  // Arrange
  const urlString = '/api/search/';
  const queryString = '2?accountNumber=123';

  const errorResponse = {
    error: { statusMessage: 'server failed' }
  };

  // Spy on private feeApiService.getRequest
  const getReqSpy = spyOn(component['feeApiService'], 'getRequest')
                      .and.returnValue(throwError(() => errorResponse));

  const errorSpy = spyOn(component, 'displayServiceErrors');
  const dispatchSpy = spyOn(component['store'], 'dispatch');

  // Act
  component.GenerateSearchAcctNumTableRows(urlString, queryString);

  // Assert
  expect(getReqSpy).toHaveBeenCalledWith(urlString + component.regionid + '/' + encodeURIComponent(queryString));
  expect(component.isData).toBeFalse();
  expect(component.loading).toBeFalse();
  expect(errorSpy).toHaveBeenCalledWith('server failed');
  expect(dispatchSpy).toHaveBeenCalledWith(jasmine.any(LOADINGACTION));
  expect(component.disbleButtonAction).toBeFalse();
});
