it('isNullOrEmpty(): should return true when value is null', () => {
  expect(component.isNullOrEmpty(null)).toBeTrue();
});

it('isNullOrEmpty(): should return true when value is undefined', () => {
  expect(component.isNullOrEmpty(undefined as any)).toBeTrue();
});

it('isNullOrEmpty(): should return true when value is empty string', () => {
  expect(component.isNullOrEmpty('')).toBeTrue();
});

it('isNullOrEmpty(): should return true when value is spaces only', () => {
  expect(component.isNullOrEmpty('   ')).toBeTrue();
});

it('isNullOrEmpty(): should return false when value contains characters', () => {
  expect(component.isNullOrEmpty('ABC')).toBeFalse();
});
