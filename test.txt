it('getControlIdComboBoxValue(): should set _controlIdComboBox.value based on matching controlId', () => {
  // Arrange
  component.controlIds = [
    { conId: 100, controlDesc: 'Control A' },
    { conId: 200, controlDesc: 'Control B' }
  ] as any;

  component.divConBraMap = { controlId: 200 } as any;
  (component as any)._controlIdComboBox = { value: null };

  // Act
  component.getControlIdComboBoxValue();

  // Assert
  expect((component as any)._controlIdComboBox.value).toEqual(['200']);
});
