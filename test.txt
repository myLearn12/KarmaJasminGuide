it('canProcessRecord(): should hide modal and call processRecord when canProceed=true', () => {
  // Arrange
  const hideSpy = jasmine.createSpy('hide');
  const processSpy = spyOn(component as any, 'processRecord');

  component['_actionModal'] = { hide: hideSpy } as any;

  // Act
  component.canProcessRecord(true);

  // Assert
  expect(hideSpy).toHaveBeenCalled();
  expect(processSpy).toHaveBeenCalled();
});

it('canProcessRecord(): should hide modal and NOT call processRecord when canProceed=false', () => {
  // Arrange
  const hideSpy = jasmine.createSpy('hide');
  const processSpy = spyOn(component as any, 'processRecord');

  component['_actionModal'] = { hide: hideSpy } as any;

  // Act
  component.canProcessRecord(false);

  // Assert
  expect(hideSpy).toHaveBeenCalled();
  expect(processSpy).not.toHaveBeenCalled();
});
