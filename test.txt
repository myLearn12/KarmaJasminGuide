it('canProcessRecord: should hide confirm modal and call processRecord when canProceed is true', () => {
  const hideSpy = spyOn(component['_confirmModal'], 'hide');
  const processSpy = spyOn(component as any, 'processRecord');

  // Act
  (component as any).canProcessRecord(true);

  // Assert
  expect(hideSpy).toHaveBeenCalled();
  expect(processSpy).toHaveBeenCalled();
});

it('canProcessRecord: should hide confirm modal and not call processRecord when canProceed is false', () => {
  const hideSpy = spyOn(component['_confirmModal'], 'hide');
  const processSpy = spyOn(component as any, 'processRecord');

  // Act
  (component as any).canProcessRecord(false);

  // Assert
  expect(hideSpy).toHaveBeenCalled();
  expect(processSpy).not.toHaveBeenCalled();
});
