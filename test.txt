it('should set showSearchTextRequiredError when value is "2"', () => {
  component._searchByRadioButton = { value: '2' } as any;
  component.selectedSearchText = '';
  spyOn(component.utils, 'isStringValid').and.returnValue(false);

  const result = component.ValidateFormControls();

  expect(component.showSearchTextRequiredError).toBeTrue();
  expect(result).toBeTrue();   // final if block
});

it('should set showSearchTextRequiredError when value is "4"', () => {
  component._searchByRadioButton = { value: '4' } as any;
  component.selectedSearchText = '';
  spyOn(component.utils, 'isStringValid').and.returnValue(false);

  const result = component.ValidateFormControls();

  expect(component.showSearchTextRequiredError).toBeTrue();
  expect(result).toBeTrue();   // final if block
});

it('should return true when showSearchByRequiredError is true', () => {
  component._searchByRadioButton = { value: null } as any;

  const result = component.ValidateFormControls();

  expect(component.showSearchByRequiredError).toBeTrue();
  expect(result).toBeTrue();   // returns from last if
});
