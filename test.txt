it('onCloneClick(): should set clone flag, set account number and redirect', () => {
  // Arrange
  component.account = { accountNbr: 'ACC123' } as any;

  component['acctService'] = {
    setCloneAccount: jasmine.createSpy('setCloneAccount'),
    setAccountNumber: jasmine.createSpy('setAccountNumber')
  } as any;

  component['utils'] = {
    redirectTo: jasmine.createSpy('redirectTo')
  } as any;

  // Act
  component.onCloneClick();

  // Assert
  expect(component['acctService'].setCloneAccount)
    .toHaveBeenCalledWith(true);

  expect(component['acctService'].setAccountNumber)
    .toHaveBeenCalledWith('ACC123');

  expect(component['utils'].redirectTo)
    .toHaveBeenCalledWith('/FeeAccrual/account/editaccount');
});
