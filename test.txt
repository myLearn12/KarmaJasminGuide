it('LoadDefaultComboBox(): error -> should call displayServiceErrors with error message', fakeAsync(() => {
  // Arrange
  const errorMessage = 'Something went wrong';
  const mockError = { error: { statusMessage: errorMessage } };

  // Spy on displayServiceErrors
  const errorSpy = spyOn(component as any, 'displayServiceErrors');

  // Mock feeApiservice.getRequest to throw error for any URL
  spyOn(component.feeApiservice, 'getRequest').and.returnValue(throwError(() => mockError));

  // Act
  (component as any).LoadDefaultComboBox();
  tick(); // flush async error

  // Assert
  expect(errorSpy).toHaveBeenCalledWith(errorMessage);
}));
