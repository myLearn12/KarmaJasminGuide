it('onClearButtonClick(): should reset search combo values, clear error message, and call GenerateTableRows', () => {
  // Arrange
  (component as any).errorBlockMessage = 'Some previous error';
  (component as any)._searchCategoryComboBox = { value: '7' };
  (component as any)._searchMinorComboBox = { value: 'AM1' };
  const generateSpy = spyOn(component as any, 'GenerateTableRows');

  // Act
  (component as any).onClearButtonClick();

  // Assert
  expect((component as any).errorBlockMessage).toBeNull();
  expect((component as any)._searchCategoryComboBox.value).toBeNull();
  expect((component as any)._searchMinorComboBox.value).toBeNull();
  expect(generateSpy).toHaveBeenCalled();
});
