it('InitialDataLoad: should set processTypeId when processType is returned', fakeAsync(() => {
  spyOn(component.feeApiservice, 'getRequest').and.returnValues(
    of({ processType: { processTypeId: 123 } }), // first API
    of(null) // second API
  );

  const dispatchSpy = spyOn(component['store'], 'dispatch');

  component.InitialDataLoad();
  tick();

  expect(component.processTypeId).toBe(123);
  expect(dispatchSpy).toHaveBeenCalledWith(new LOADINGACTION(false));
}));
