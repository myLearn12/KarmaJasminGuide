it('canProcessRecord: should hide modal and call processRecord when canProceed = true', () => {
  const hideSpy = spyOn((component as any)._actionModal, 'hide');
  const procSpy = spyOn(component as any, 'processRecord');

  (component as any).canProcessRecord(true);

  expect(hideSpy).toHaveBeenCalled();
  expect(procSpy).toHaveBeenCalled();
});
