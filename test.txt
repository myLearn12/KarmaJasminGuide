it('InitialDataLoad: should call displayServiceErrors on error', fakeAsync(() => {
  const errMsg = 'Something went wrong';
  spyOn(component.feeApiservice, 'getRequest').and.returnValue(
    throwError({ error: { statusMessage: errMsg } })
  );
  const errSpy = spyOn(component as any, 'displayServiceErrors');

  component.InitialDataLoad();
  tick();

  expect(errSpy).toHaveBeenCalledWith(errMsg);
}));
