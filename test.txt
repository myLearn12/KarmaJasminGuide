it('revShareDivMappingMessage: sets division NOT mapped message when no revShareDesc', () => {
  (component as any).cbDivision = { value: '20' };
  (component as any).cbAltDivision = { value: '30' };

  component.divisions = [
    { divisionId: 20, divisionCd: 'D20' },
    { divisionId: 30, divisionCd: 'D30' }
  ] as any;

  component.revShareDivMapping = [
    { divisionId: 30, revShareDesc: 'SomeAltDesc' } // only alt mapped, division missing
  ] as any;

  spyOn(component.utils, 'isStringValid').and.callFake((s: any) => !!s);

  component.revShareDivMappingMessage();

  expect(component.divCdMapMessage).toBe('Division Cd D20 is not mapped.');
});
