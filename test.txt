// ✅ when divisionCd is "0000" → clears divisionCd, divisionDesc, and cbDivision.value
it('SetunassignedOrArchive() clears division for unassigned ("0000")', () => {
  const utilsStub = { CreateRequiredFormControl: () => ({}), CreateDateFormControl: () => ({}) };
  const storeMock = { pipe: jasmine.createSpy('pipe').and.returnValue(of(false)), dispatch: () => {} };

  const c = new EditaccountComponent(
    {} as any,     // http
    {} as any,     // appconfig
    storeMock as any,
    utilsStub as any,
    {} as any      // acctService
  );

  // minimal state used by the method
  c.account = { divisionCd: '0000', divisionDesc: 'Some Division' } as any;
  c.cbDivision = { value: ['10'] } as any;

  // act
  c.SetunassignedOrArchive();

  // assert
  expect(c.account.divisionCd).toBeNull();
  expect(c.account.divisionDesc).toBeNull();
  expect(c.cbDivision.value).toBeNull();
});
