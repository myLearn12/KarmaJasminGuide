it('onSearchButtonClick() - searchByValue=4 should dispatch loading and call GenerateSearchAcctNameTableRows with longAccountNbr', () => {
  // arrange
  component.ValidateFormControls = jasmine.createSpy().and.returnValue(false);

  // make the radio selection "4"
  component._searchByRadioButton = { value: '4' };

  component.selectedSearchText = '  1234567890  '; // trimmed in code
  component.appConfigService = { getAllAccountExportUrl: '/api/accounts/' };

  spyOn(component, 'GenerateSearchAcctNameTableRows').and.callFake(() => {});

  // store is private; spy it via (component as any).store
  const storeSpy = spyOn((component as any).store, 'dispatch').and.callFake(() => {});

  // act
  component.onSearchButtonClick();

  // assert
  expect(component.loading).toBeTrue();

  expect(storeSpy).toHaveBeenCalledWith(new LOADINGACTION(true));

  const expectedUrl = '/api/accounts/';
  const expectedQuery = '2?longAccountNbr=' + encodeURIComponent(component.selectedSearchText.trim());

  expect(component.GenerateSearchAcctNameTableRows)
    .toHaveBeenCalledWith(expectedUrl, expectedQuery);
});
