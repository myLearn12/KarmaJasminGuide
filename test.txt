it('setIsRevShareValid(): invalid revshare should set isRevShareValid = false', () => {
  component.account = { revshare: 'abc' }; // not numeric

  spyOn(component['utils'], 'isStringValid').and.returnValue(true);

  (component as any).setIsRevShareValid();

  expect(component.isRevShareValid).toBeFalse();
});

it('setIsRevShareValid(): valid decimal revshare should set isRevShareValid = true', () => {
  component.account = { revshare: '0.5' };

  spyOn(component['utils'], 'isStringValid').and.returnValue(true);

  (component as any).setIsRevShareValid();

  expect(component.isRevShareValid).toBeTrue();
});

it('setIsRevShareValid(): valid integer revshare should set isRevShareValid = true', () => {
  component.account = { revshare: '1' };

  spyOn(component['utils'], 'isStringValid').and.returnValue(true);

  (component as any).setIsRevShareValid();

  expect(component.isRevShareValid).toBeTrue();
});

it('setIsRevShareValid(): one decimal place should convert toFixed(2) and set true', () => {
  component.account = { revshare: '0.7' };

  spyOn(component['utils'], 'isStringValid').and.returnValue(true);

  (component as any).setIsRevShareValid();

  expect(component.isRevShareValid).toBeTrue();
  expect(component.account.revshare).toBe('0.70'); // formatted
});
