it('getRevenueShareComboBoxValue: should set value when revShareTypeId matches', () => {
  // Arrange
  component.revenueShares = [
    { revShareTypeId: 10, revShareTypeDesc: 'Revenue A', revShareTypeCd: 'R1' } as any,
    { revShareTypeId: 20, revShareTypeDesc: 'Revenue B', revShareTypeCd: 'R2' } as any
  ];
  component.revShrdivmap = { revShareTypeId: 10 } as any;

  (component as any)._revenueShareComboBox = { value: null };

  // Act
  component.getRevenueShareComboBoxValue();

  // Assert
  expect((component as any)._revenueShareComboBox.value).toEqual(['10']);
});
