it('InitialDataLoad(): when API errors -> sets isData=false, calls displayServiceErrors, clears loading, dispatches LOADINGACTION(false) and resets disbleButtonAction', fakeAsync(() => {
  // Arrange
  const fakeError = { error: { statusCode: '1', statusMessage: 'server failed' } };

  // stub feeApiservice.getRequest to return an error observable
  component['feeApiservice'] = {
    getRequest: (url: string) => throwError(() => fakeError)
  } as any;

  // stub required appConfigService value used inside the method
  component['appConfigService'] = {
    getAccountByStatusurl: '/api/accounts/status'
  } as any;

  // some method uses regionid to build URL â€” set one
  component.regionid = 'R001';

  // spies for side-effects
  const displaySpy = spyOn(component as any, 'displayServiceErrors');
  component['store'] = { dispatch: jasmine.createSpy('dispatch') } as any;

  // set initial flags as the method expects
  component.loading = true;
  component.disbleButtonAction = true;
  component.isData = true; // ensure it flips to false on error

  // Act
  component.InitialDataLoad();
  tick(); // flush observable error

  // Assert
  expect(component.isData).toBeFalse();
  expect(displaySpy).toHaveBeenCalledWith('server failed');
  expect(component.loading).toBeFalse();
  expect(component['store'].dispatch).toHaveBeenCalledWith(jasmine.any(LOADINGACTION));
  expect(component.disbleButtonAction).toBeFalse();
}));
