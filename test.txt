it('InitialDataLoad: should set infoBlockMessage when status = INPROGRESS', fakeAsync(() => {
  spyOn(component.feeApiservice, 'getRequest').and.returnValues(
    of({ processType: { processTypeId: 321 } }), // first API
    of({ status: 'INPROGRESS', creatById: 'u1', creatDt: '2025-09-01' }) // second API
  );
  spyOn(component['utils'], 'getDisplayDate').and.returnValue('09/01/2025');

  component.InitialDataLoad();
  tick();

  expect(component.infoBlockMessage).toContain('Load Cash Fees is INPROGRESS');
  expect(component.infoBlockMessage).toContain('u1');
}));
