it('GenerateTableRows() - error: sets isData=false, calls displayServiceErrors, clears loading and dispatches LOADINGACTION(false)', () => {
  // arrange
  const feeApi = TestBed.inject(FeeApiService) as any;
  const store = TestBed.inject(Store) as any;

  // ensure appConfig URL used in the method is defined (screenshot shows: + '0' + '/2')
  (component as any).appConfigService = { getAllAccountExportUrl: '/api/accounts' } as any;

  // make sure subscription exists so .add(...) doesn't throw
  (component as any).subscription = new Subscription();

  // stub the API to return an error
  spyOn(feeApi, 'getRequest').and.returnValue(throwError({ error: { statusMessage: 'server failed' } }));

  // spy displayServiceErrors and store.dispatch
  spyOn(component as any, 'displayServiceErrors').and.callThrough();
  spyOn(store, 'dispatch').and.callFake(() => {});

  // act
  (component as any).GenerateTableRows();

  // assert
  expect(feeApi.getRequest).toHaveBeenCalledWith('/api/accounts' + '0' + '/2');
  expect((component as any).isData).toBeFalse();
  expect((component as any).loading).toBeFalse();
  expect((component as any).displayServiceErrors).toHaveBeenCalledWith('server failed');
  expect(store.dispatch).toHaveBeenCalled();
});
