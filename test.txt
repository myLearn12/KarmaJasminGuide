it('OnStatusChange: when status is "2" → clears dateclosed and hides closed-required error; sets open-required if dateopen invalid', fakeAsync(() => {
  // Arrange
  const statusRef = { value: ['2'] }; // ['2'] → .toString() === '2'
  (component as any).cbStatus = statusRef;
  (component as any).cbstatus = statusRef;

  component.account = { dateopen: '', dateclosed: '2025-08-10' } as any;

  // isStringValid('') -> false  => showAcctOpenRequiredError should become true
  spyOn(component['utils'], 'isStringValid').and.returnValue(false);

  // Act
  (component as any).OnStatusChange();
  tick();

  // Assert
  expect(component.statusRequired).toBeFalse();           // value not null
  expect(component.account.dateclosed).toBeNull();        // cleared
  expect(component.showAcctClosedRequiredError).toBeFalse();

  // dateopen invalid -> open required error true
  expect(component.showAcctOpenRequiredError).toBeTrue();
}));
