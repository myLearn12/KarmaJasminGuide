it('onUpdateModalClick(): else branch should deep clone revShrGlAcctmap, call helpers and show modal', () => {
  // Arrange
  component.isRowSelected = true;
  component.revShrGlAcctmap = { revShareGlAcctId: 1, revShareDesc: 'Test' } as any;

  spyOn(component as any, 'setIntialvalues');
  spyOn(component as any, 'getRevenueShareComboBoxValue');
  spyOn(component as any, 'getCostCenterComboBoxValue');
  (component as any)._actionModal = { show: jasmine.createSpy('show') };

  // Act
  component.onUpdateModalClick();

  // Assert deep clone
  expect(component.selectedRevShrGlAcctMap).toEqual(component.revShrGlAcctmap);
  expect(component.selectedRevShrGlAcctMap).not.toBe(component.revShrGlAcctmap);

  // Assert helper calls + modal
  expect((component as any).setIntialvalues).toHaveBeenCalled();
  expect((component as any).getRevenueShareComboBoxValue).toHaveBeenCalled();
  expect((component as any).getCostCenterComboBoxValue).toHaveBeenCalled();
  expect((component as any)._actionModal.show).toHaveBeenCalled();
});
