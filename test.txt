it('should execute error branch in GenerateSearchAcctNameTableRows()', () => {
  // Arrange
  const url = '/api/search/';
  const query = 'abc';

  const errorResponse = {
    error: { statusMessage: 'server failed' }
  };

  // spy on displayServiceErrors
  const displaySpy = spyOn(component, 'displayServiceErrors');

  // spy on store.dispatch
  const storeSpy = spyOn(component['store'], 'dispatch');

  // spy on feeApiService.getRequest (private â†’ access through bracket notation)
  spyOn(component['feeApiService'], 'getRequest')
    .and.returnValue(throwError(() => errorResponse));

  // Act
  component.GenerateSearchAcctNameTableRows(url, query);

  // Assert
  expect(component.isData).toBeFalse();
  expect(displaySpy).toHaveBeenCalledWith('server failed');
  expect(component.loading).toBeFalse();
  expect(storeSpy).toHaveBeenCalledWith(new LOADINGACTION(false));
  expect(component.disbleButtonAction).toBeFalse();
});
