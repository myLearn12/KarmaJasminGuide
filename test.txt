it('onPcaChange: should set isPcaValid = !isStringAlphaNumeric when pcanumber is valid', fakeAsync(() => {
  // Arrange
  component.account = { pcanumber: 'ABC123' } as any;

  spyOn(component['utils'], 'isStringValid').and.returnValue(true);
  spyOn(component['utils'], 'isStringAlphaNumeric').and.returnValue(true); // returns true -> !true = false

  // Act
  (component as any).onPcaChange();
  tick();

  // Assert
  expect(component.isPcaValid).toBeFalse();

  // Now flip: if isStringAlphaNumeric false -> !false = true
  (component as any).isPcaValid = null as any; // reset
  (component['utils'].isStringAlphaNumeric as jasmine.Spy).and.returnValue(false);

  (component as any).onPcaChange();
  tick();

  expect(component.isPcaValid).toBeTrue();
}));
