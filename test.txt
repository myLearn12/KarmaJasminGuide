it('InitialDataLoad() â†’ error handler sets error message, disables save, stops loading, and dispatches action', () => {
  const utilsStub = {
    CreateRequiredFormControl: () => ({}),
    CreateDateFormControl: () => ({}),
    LoadComboBoxData: () => [],
  };
  const acctServiceStub = {
    LoadOfficerData: () => [],
    loadStatusData: () => [],
  };
  const storeMock = { dispatch: jasmine.createSpy('dispatch') };

  const component = new EditaccountComponent(
    {} as any,           // http
    {} as any,           // appConfig
    storeMock as any,    // store
    utilsStub as any,    // utils
    acctServiceStub as any // acctService
  );

  // force error in forkJoin
  component['feeApiservice'] = {
    getRequest: () => throwError(() => ({ error: { statusMessage: 'fail-msg' } })),
  } as any;

  // spy error display
  component['displayServiceErrors'] = jasmine.createSpy('displayServiceErrors');

  // act
  component['InitialDataLoad']();

  // assert error branch effects
  expect(component['displayServiceErrors']).toHaveBeenCalledWith('fail-msg');
  expect(component.disableSavebtn).toBeTrue();
  expect(component.loading).toBeFalse();
  expect(storeMock.dispatch).toHaveBeenCalled(); // LOADINGACTION(false)
});
