it('GenerateTableRows: should handle error by calling displayServiceErrors, reset loading, and dispatch LOADINGACTION(false)', fakeAsync(() => {
  // Arrange
  // make getRequest throw an error
  const errorResponse = { error: { statusMessage: 'Test error' } };
  spyOn(component.feeApiservice, 'getRequest').and.returnValue(throwError(() => errorResponse));

  const errorSpy = spyOn(component as any, 'displayServiceErrors');
  const dispatchSpy = spyOn(component['store'], 'dispatch');

  // Act
  component.GenerateTableRows();
  tick(); // flush the observable

  // Assert
  expect(errorSpy).toHaveBeenCalledWith('Test error');
  expect(component.loading).toBeFalse();
  expect(dispatchSpy).toHaveBeenCalledWith(new LOADINGACTION(false));
}));
