it('onClearButtonClick(): should reset search combo values, clear error, and regenerate table rows', () => {
  // Arrange
  component.errorBlockMessage = 'Some error';
  (component as any)._searchDivisionComboBox = { value: '10' };
  (component as any)._searchControlIdComboBox = { value: '200' };
  (component as any)._searchBranchCdComboBox = { value: 'B001' };

  const generateSpy = spyOn(component as any, 'GenerateTableRows');

  // Act
  component.onClearButtonClick();

  // Assert
  expect(component.errorBlockMessage).toBeNull();
  expect((component as any)._searchDivisionComboBox.value).toBeNull();
  expect((component as any)._searchControlIdComboBox.value).toBeNull();
  expect((component as any)._searchBranchCdComboBox.value).toBeNull();
  expect(generateSpy).toHaveBeenCalled();
});
