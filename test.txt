it('processRecord(): server error -> calls displayServiceErrors, clears loading and dispatches LOADINGACTION(false)', () => {
  // arrange
  const uploadList = [{ acctNbr: 'A3', miscProceedsAmt: 50, paidThruDt: '2025-11-30' }];
  component.sheetName = 'sheet1';
  component.worksheet = { [component.sheetName]: uploadList };
  component.region = { regionId: 'R003' };

  const serverError = { error: { statusMessage: 'server failed' } };
  spyOn(component.feeApiService, 'postRequest').and.returnValue(throwError(serverError));
  const displaySpy = spyOn(component, 'displayServiceErrors');
  const dispSpy = spyOn(component.store, 'dispatch');

  // act
  component.processRecord();
  fixture.detectChanges();

  // assert
  expect(displaySpy).toHaveBeenCalledWith('server failed');
  expect(component.loading).toBeFalse();
  expect(dispSpy).toHaveBeenCalled();
})
