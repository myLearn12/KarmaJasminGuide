it('should set _regionComboBox.value with matching region in getRegionComboBoxValue()', () => {
  // Arrange
  component.regions = [
    { regionId: 1, regionDescription: 'Region 1' },
    { regionId: 2, regionDescription: 'Region 2' }
  ];

  component.unassignedaccountmap = { regId: 2 };

  // mock _regionComboBox
  (component as any)._regionComboBox = { value: null };

  // Act
  component.getRegionComboBoxValue();

  // Assert
  expect((component as any)._regionComboBox.value).toEqual([
    { regionId: 2, regionDescription: 'Region 2' }
  ]);
});
