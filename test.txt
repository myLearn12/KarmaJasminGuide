it('onClearButtonClick: should reset error message, clear search combos and regenerate table rows', () => {
  // Arrange
  component.errorBlockMessage = 'some error';
  (component as any).searchRevenueShareComboBox = { value: '10' };
  (component as any)._searchDivisionComboBox = { value: '20' };

  const genSpy = spyOn(component as any, 'GenerateTableRows').and.stub();

  // Act
  component.onClearButtonClick();

  // Assert
  expect(component.errorBlockMessage).toBeNull();
  expect((component as any).searchRevenueShareComboBox.value).toBeNull();
  expect((component as any)._searchDivisionComboBox.value).toBeNull();
  expect(genSpy).toHaveBeenCalled();
});
