it('getDivisionComboBoxVal: should set value when divisionId matches', () => {
  // Arrange
  component.divisions = [
    { divisionId: 20, divisionDesc: 'Division A', divisionCd: 'D1' } as any,
    { divisionId: 30, divisionDesc: 'Division B', divisionCd: 'D2' } as any
  ];
  component.revShrdivmap = { divisionId: 20 } as any;

  (component as any)._divisionComboBox = { value: null };

  // Act
  component.getDivisionComboBoxVal();

  // Assert
  expect((component as any)._divisionComboBox.value).toEqual(['20']);
});

it('getDivisionComboBoxVal: should set empty array when no divisionId matches', () => {
  // Arrange
  component.divisions = [
    { divisionId: 40, divisionDesc: 'Division C', divisionCd: 'D3' } as any
  ];
  component.revShrdivmap = { divisionId: 99 } as any;

  (component as any)._divisionComboBox = { value: null };

  // Act
  component.getDivisionComboBoxVal();

  // Assert
  expect((component as any)._divisionComboBox.value).toEqual([]);
});
