describe('canProcessRecord()', () => {

  it('should hide confirm modal and call processRecord when canProceed = true', () => {
    spyOn(component.confirmModal, 'hide');
    spyOn(component, 'processRecord');

    component.canProcessRecord(true);

    expect(component.confirmModal.hide).toHaveBeenCalled();
    expect(component.processRecord).toHaveBeenCalled();
  });

  it('should hide confirm modal and NOT call processRecord when canProceed = false', () => {
    spyOn(component.confirmModal, 'hide');
    spyOn(component, 'processRecord');

    component.canProcessRecord(false);

    expect(component.confirmModal.hide).toHaveBeenCalled();
    expect(component.processRecord).not.toHaveBeenCalled();
  });

});
