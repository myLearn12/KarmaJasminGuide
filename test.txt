it('onActionModalClick: should hide modal, set selectedFeeType and call processData', () => {
  // Arrange
  (component as any)._EstateCashFeeModal = { hide: jasmine.createSpy('hide') };
  spyOn(component, 'processData');
  
  // Act
  component.onActionModalClick('CashFee');
  
  // Assert
  expect((component as any)._EstateCashFeeModal.hide).toHaveBeenCalled();
  expect(component.selectedFeeType).toBe('CashFee');
  expect(component.processData).toHaveBeenCalled();
});
