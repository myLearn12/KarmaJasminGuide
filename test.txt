it('onOffcrNumberChange(): when cbOfficer.value is null -> sets officerRequired true, updates selectedofficerNm and calls setIsOfcrNmbrValid', () => {
  // arrange
  component.cbOfficer = { value: null } as any;
  component.fcn_OffcrNumber = { value: 'O1' } as any;
  component.officers = { O1: { officerNm: 'Bob' } } as any;
  spyOn(component, 'setIsOfcrNmbrValid');

  // act
  component.onOffcrNumberChange();

  // assert
  expect(component.officerRequired).toBeTrue();
  expect(component.selectedofficerNm).toBe('Bob');
  expect(component.setIsOfcrNmbrValid).toHaveBeenCalled();
});

it('onOffcrNumberChange(): when cbOfficer.value is NOT null -> sets officerRequired false, updates selectedofficerNm and calls setIsOfcrNmbrValid', () => {
  // arrange
  component.cbOfficer = { value: 'some-value' } as any;
  component.fcn_OffcrNumber = { value: 'O2' } as any;
  component.officers = { O2: { officerNm: 'Alice' } } as any;
  spyOn(component, 'setIsOfcrNmbrValid');

  // act
  component.onOffcrNumberChange();

  // assert
  expect(component.officerRequired).toBeFalse();
  expect(component.selectedofficerNm).toBe('Alice');
  expect(component.setIsOfcrNmbrValid).toHaveBeenCalled();
});
