it('onSearchButtonClick(): when no matches -> shows warning, clears search combos and calls GenerateTableRows', () => {
  component.initialDivConBraMaps = [
    { divisionId: 1, controlId: 1, branchCd: 'X' }
  ] as any[];

  (component as any)._searchDivisionComboBox = { value: '99' };
  (component as any)._searchControlIdComboBox = { value: '999' };
  (component as any)._searchBranchCdComboBox = { value: 'ZZ' };

  const warnSpy = spyOn(component as any, 'SetMessageTypeAndValue');
  const genSpy = spyOn(component as any, 'GenerateTableRows');

  component.onSearchButtonClick();

  // Warn shown with searchOptionMessage and GenerateTableRows called
  expect(warnSpy).toHaveBeenCalledWith(jasmine.anything(), component.searchOptionMessage);
  expect(genSpy).toHaveBeenCalled();

  // Search combos reset to null
  expect((component as any)._searchDivisionComboBox.value).toBeNull();
  expect((component as any)._searchControlIdComboBox.value).toBeNull();
  expect((component as any)._searchBranchCdComboBox.value).toBeNull();
});
