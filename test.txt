it('getRegionComboBoxValue(): should set regionComboBox.value based on matching regionId', () => {
  // Arrange
  component.regions = [
    { regionId: 10, regionDescription: 'East' },
    { regionId: 20, regionDescription: 'West' }
  ] as any;

  component.regdivmap = { regionId: 20 } as any;

  component.regionComboBox = { value: null } as any;

  // Act
  component.getRegionComboBoxValue();

  // Assert
  expect(component.regionComboBox.value).toEqual(['20']);
});
