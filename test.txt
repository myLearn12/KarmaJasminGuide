it('onOffNumberChange(): else branch sets isOfficerNumValid = false', () => {
  // arrange
  spyOn(component['utils'], 'isStringValid').and.returnValue(false);
  component.account = { officer: null } as any;

  // act
  component.onOffNumberChange();

  // assert
  expect(component.isOfficerNumValid).toBeFalse();
});
