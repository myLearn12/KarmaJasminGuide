it('should handle searchByValue=1 with invalid string and call GenerateSearchAcctNameTableRows("2")', () => {
  component._searchByRadioButton = { value: '1' } as any;
  component.selectedSearchText = '';
  spyOn(utils, 'isStringValid').and.returnValue(false);

  spyOn(component, 'GenerateSearchAcctNameTableRows');

  component.onSearchButtonClick();

  expect(component.GenerateSearchAcctNameTableRows)
    .toHaveBeenCalledWith(appConfigService.getAccountByStatusurl, '2');
});
