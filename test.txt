it('InitialExcelDataLoad(): searchByValue 3 → invalid text → queryString = region/1', () => {
  const component = new ViewaccountsComponent(null, null, null, null);

  component.regionid = "R001";

  component._searchByRadioButton = { value: "3" } as any;
  component.selectedSearchText = "";

  component.utils = {
    isStringValid: jasmine.createSpy().and.returnValue(false)
  } as any;

  component.InitialExcelDataLoad();

  expect(component.utils.isStringValid).toHaveBeenCalledWith("");
  // branch covered: queryString = regionid + "/1"
});
