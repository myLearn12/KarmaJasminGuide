it('TaxAccountChange: when rbnTax.value is true should set action=1, Tax=true, NonTax=false', () => {
  // Arrange
  component.rbnTax = { value: true, setValue: jasmine.createSpy('setValue') };
  component.rbnNonTax = { setValue: jasmine.createSpy('setValue') };
  component._actionRadioButton = { value: null };

  // Act
  component.TaxAccountChange();

  // Assert
  expect(component._actionRadioButton.value).toBe("1");
  expect(component.rbnTax.setValue).toHaveBeenCalledWith(true);
  expect(component.rbnNonTax.setValue).toHaveBeenCalledWith(false);
});

it('TaxAccountChange: when rbnTax.value is false should set action=2, NonTax=true, Tax=false', () => {
  // Arrange
  component.rbnTax = { value: false, setValue: jasmine.createSpy('setValue') };
  component.rbnNonTax = { setValue: jasmine.createSpy('setValue') };
  component._actionRadioButton = { value: null };

  // Act
  component.TaxAccountChange();

  // Assert
  expect(component._actionRadioButton.value).toBe("2");
  expect(component.rbnNonTax.setValue).toHaveBeenCalledWith(true);
  expect(component.rbnTax.setValue).toHaveBeenCalledWith(false);
});
