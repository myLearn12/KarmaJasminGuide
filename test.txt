it('InitialDataLoad(): next branch - calls LoadRegionData and dispatches LOADINGACTION(false)', () => {
  // Arrange - ensure required private/protected properties exist
  (component as any).regionid = 123;
  (component as any).appConfigService = {
    getRegionByIdUrl: '/regionById/',
    getProcessControlConfigByType: '/procByType/'
  };
  // spyable service stubs
  const feeApiSpy = jasmine.createSpyObj('feeApiService', ['getRequest']);
  // region endpoint -> object
  feeApiSpy.getRequest.and.callFake((url: string) => {
    if (url.indexOf('/regionById/') === 0 || url.includes('/regionById/')) {
      return of({ regionId: 1, regionDescription: 'R1' });
    }
    // processType endpoint -> object having processType key
    if (url.indexOf('/procByType/') === 0 || url.includes('/procByType/')) {
      return of({ processType: { processTypeId: 42 } });
    }
    return of([] as any);
  });
  (component as any).feeApiservice = feeApiSpy;

  // store stub and spies
  const storeSpy = jasmine.createSpyObj('store', ['dispatch']);
  (component as any).store = storeSpy;

  // spy LoadRegionData so we can assert it was called with region
  spyOn(component as any, 'LoadRegionData').and.callThrough();

  // Act
  (component as any).InitialDataLoad();

  // Assert
  expect((component as any).LoadRegionData).toHaveBeenCalledWith(jasmine.objectContaining({ regionId: 1, regionDescription: 'R1' }));
  expect((component as any).processTypeId).toBe(42);
  expect(storeSpy.dispatch).toHaveBeenCalled(); // LOADINGACTION(false) dispatched at least once
});
