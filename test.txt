it('onDivComboBoxChangeEvent(): sets showDivRequiredError true when division control has required error', () => {
  // arrange: stub the control so hasError('required') returns true
  (component as any).fcn_divisionDesc = {
    hasError: (err: string) => err === 'required'
  };

  component.showDivRequiredError = false; // initial

  // act
  (component as any).onDivComboBoxChangeEvent();

  // assert
  expect(component.showDivRequiredError).toBeTrue();
});

it('onDivComboBoxChangeEvent(): sets showDivRequiredError false when division control has no required error', () => {
  // arrange: stub the control so hasError('required') returns false
  (component as any).fcn_divisionDesc = {
    hasError: (_: string) => false
  };

  component.showDivRequiredError = true; // initial

  // act
  (component as any).onDivComboBoxChangeEvent();

  // assert
  expect(component.showDivRequiredError).toBeFalse();
});
