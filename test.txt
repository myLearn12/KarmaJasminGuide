it('getBranchCdComboBoxValue(): should set _branchCdComboBox.value based on matching branchCd', () => {
  // Arrange
  component.branchCds = [
    { branchCode: 'B001', branchDesc: 'Branch One' },
    { branchCode: 'B002', branchDesc: 'Branch Two' }
  ] as any;

  component.divConBraMap = { branchCd: 'B002' } as any;
  (component as any)._branchCdComboBox = { value: null };

  // Act
  component.getBranchCdComboBoxValue();

  // Assert
  expect((component as any)._branchCdComboBox.value).toEqual(['B002']);
});
