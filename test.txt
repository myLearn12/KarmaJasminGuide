it('getDivisionComboBoxVal(): should set minorComboBox.value when matching acctMnrTypeCode found', () => {
  // Arrange
  (component as any).minorComboBox = { value: null };
  component.accountMinors = [
    { acctMnrTypeCode: 'AM1', acctMnrTypeDesc: 'Minor A' },
    { acctMnrTypeCode: 'AM2', acctMnrTypeDesc: 'Minor B' }
  ] as any[];
  component.catMnrmap = { acctMnrTypeCd: 'AM1' } as any;

  // Act
  (component as any).getDivisionComboBoxVal();

  // Assert
  expect((component as any).minorComboBox.value).toEqual(['AM1']);
});
