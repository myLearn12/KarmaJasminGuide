it('getDivisionComboBoxVal(): should set _divisionComboBox.value based on matching divisionId', () => {
  // Arrange
  component.divisions = [
    { divisionId: 10, divisionCd: 'D10', divisionDesc: 'Division 10' },
    { divisionId: 20, divisionCd: 'D20', divisionDesc: 'Division 20' }
  ] as any;

  component.divConBraMap = { divisionId: 20 } as any;
  (component as any)._divisionComboBox = { value: null };

  // Act
  component.getDivisionComboBoxVal();

  // Assert
  expect((component as any)._divisionComboBox.value).toEqual(['20']);
});
