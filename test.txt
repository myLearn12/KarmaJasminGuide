it('processRecord(): success branch computes totals when NO error records and does NOT show action modal', () => {
  // required inputs
  component.worksheet = { sheet1: [{ acct: 'A1' }] } as any;
  component.sheetName = 'sheet1';
  component.accoutColumnHeader = 'acct';
  component.region = { regionId: 1 } as any;
  component.processTypeId = 10;
  component.updateAccounts = [{ hasError: false }];
  component._actionModal = { show: jasmine.createSpy('show') } as any;

  // FIX: avoid undefined error
  component.appConfigService = { accrualUpdateUrl: 'URL/' } as any;

  spyOn(component.utils, 'ToEstDate').and.returnValue('DATE');
  spyOn(component.utils, 'getCurrentUser').and.returnValue('USER');
  spyOn(component.store, 'dispatch');
  spyOn(component, 'LoadTableRows');
  spyOn(component, 'clearSelection');

  feeApiService.postRequest.and.returnValue(of({}));

  component.processRecord();

  expect(component.loading).toBeFalse();
  expect(component.store.dispatch).toHaveBeenCalledWith(new LOADINGACTION(false));
  expect(component.LoadTableRows).toHaveBeenCalled();
  expect(component._actionModal.show).not.toHaveBeenCalled();
});
