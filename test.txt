it('onSaveButtonClick: should reset and show error message on API error', fakeAsync(() => {
  // Arrange
  component.account = { acctnum: 'A1', sticurrent: 123 } as any;
  (component as any).appConfigService = { updateManualFundFeePostingUrl: '/mock-url/' };

  const mockError = { error: { statusMessage: 'Update failed' } };
  spyOn(component.feeApiservice, 'putRequest').and.returnValue(throwError(() => mockError));
  const resetSpy = spyOn(component as any, 'resetData');
  const errorSpy = spyOn(component as any, 'displayServiceErrors');
  const dispatchSpy = spyOn(component.store, 'dispatch');

  // Act
  (component as any).onSaveButtonClick();
  tick();

  // Assert
  expect(resetSpy).toHaveBeenCalled();
  expect(errorSpy).toHaveBeenCalledWith('Update failed');
  expect(component.loading).toBeFalse();
  expect(dispatchSpy).toHaveBeenCalledWith(jasmine.any(LOADINGACTION));
}));
