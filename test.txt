it('OnStatusChange: should set showAcctOpenRequiredError=true when dateopen is invalid', fakeAsync(() => {
  // Arrange
  (component as any).cbStatus = { value: ['1'] };   // dummy status to avoid null
  (component as any).cbstatus = { value: ['1'] };

  component.account = { dateopen: '' } as any; // empty dateopen
  spyOn(component['utils'], 'isStringValid').and.returnValue(false);

  // Act
  (component as any).OnStatusChange();
  tick();

  // Assert
  expect(component.showAcctOpenRequiredError).toBeTrue();
}));
