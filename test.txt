it('ValidateFormData: should set errors when Paid-Thru date is more than 10 years in the future', () => {
  const futureDate = new Date();
  futureDate.setFullYear(futureDate.getFullYear() + 11); // 11 years in future

  component.account = { formulaCd: 1, datepdthru: futureDate.toISOString() } as any;

  spyOn(component.utils, 'ToDate').and.callFake((d?: any) => d ? new Date(d) : new Date());
  spyOn(component.utils, 'isStringValid').and.returnValue(true);

  component.ValidateFormData();

  expect(component.firstValidationErrorMessage)
    .toBe("The Paid-Thru date is invalid because it's more than 10 years in the future.");
  expect(component.secondValidationErrorMessage)
    .toContain("check your computer's system date");
  expect(component.thirdValidationErrorMessage)
    .toBe("Press Close to re-enter.");
});
