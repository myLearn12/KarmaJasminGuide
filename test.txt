it('InitialDataLoad: should slice error message containing 14th business day', fakeAsync(() => {
  spyOn(component.feeApiservice, 'getRequest').and.returnValue(
    throwError({ error: { statusMessage: 'ERR: 14th business day cutoff' } })
  );

  component.InitialDataLoad();
  tick();

  expect(component.infoBlockMessage).toBe('14th business day cutoff');
}));
