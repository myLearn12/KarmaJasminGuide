it('getcatMnrmapComboBoxValue(): sets _categoryComboBox.value when a matching category exists', () => {
  // Arrange
  (component as any)._categoryComboBox = { value: null };
  component.categories = [
    { categoryId: 7, categoryDescription: 'Estate', categoryCode: '4' },
    { categoryId: 8, categoryDescription: 'Other',  categoryCode: '7' }
  ] as any[];
  component.catMnrmap = { catId: 7 } as any;

  // Act
  (component as any).getcatMnrmapComboBoxValue();

  // Assert
  expect((component as any)._categoryComboBox.value).toEqual(['7']);
});
