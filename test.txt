it('onSaveButtonClick: should call processData when formulaCd=1 and cash > 0', () => {
  component.account = { catCd: '7', cashfee: 50, cashpp: 0, formulacd: 1 } as any;

  spyOn(component['_EstateCashFeeModal'], 'show');
  spyOn(component, 'processData');
  spyOn(component, 'validateFormControls').and.returnValue(false);
  spyOn(component, 'validateFormData').and.returnValue(false);
  spyOn(component['utils'], 'getFormatedNumber').and.callFake((v: any) => v);

  component.onSaveButtonClick();

  expect(component.processData).toHaveBeenCalled();
  expect(component['_EstateCashFeeModal'].show).not.toHaveBeenCalled();
});
