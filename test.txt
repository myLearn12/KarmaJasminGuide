it('onSearchButtonClick(): when control+branch selected -> loads filtered rows by control+branch', () => {
  component.initialDivConBraMaps = [
    { divisionId: 10, controlId: 100, branchCd: 'B1' },
    { divisionId: 11, controlId: 200, branchCd: 'B2' },
    { divisionId: 12, controlId: 200, branchCd: 'B2' }
  ] as any[];

  (component as any)._searchDivisionComboBox = { value: null };
  (component as any)._searchControlIdComboBox = { value: '200' };
  (component as any)._searchBranchCdComboBox = { value: 'B2' };

  const loadSpy = spyOn(component as any, 'LoadTableRows');

  component.onSearchButtonClick();

  expect(loadSpy).toHaveBeenCalled();
  const rows = loadSpy.calls.mostRecent().args[0];
  expect(rows.length).toBe(2);
  expect(rows.every(r => r.controlId.toString() === '200' && r.branchCd === 'B2')).toBeTrue();
});
